 * 贪心就行,n为奇数时，要特别注意，只能转换为0或者1中个数多的，否则会出错
 * n为偶数就没这个限制

#include <bits/stdc++.h>
using namespace std;
void solve(){
    string str;
    cin >> str;
    int n = str.size();
    string s1,s2;
    int cnt1 = 0, cnt0 = 0;
    for (int i = 0; i < n; ++i) {
        if (i%2) {
            s1 += '0';
            s2 += '1';
        }else {
            s1 += '1';
            s2 += '0';
        }
        if (str[i] == '1')
            cnt1++;
        else
            cnt0++;
    }
    auto cal = [&] (string s, string target) ->int {
        int ans = 0;
        for (int i = 0; i < n; ++i) {
            if (s[i] != target[i]) {
                for (int j = i + 1; j < n; ++j) {
                    if (s[j] == target[i]) {
                        for (int k = j - 1; k >= i; --k) {
                            ++ans;
                            swap(s[k], s[k + 1]);
                        }
                        break;
                    }
                }
            }
        }
        return ans;
    };
    if (n%2) {
        if (cnt1 > cnt0)
            cout << cal(str, s1) << endl;
        else
            cout << cal(str, s2) << endl;
    }
    else
        cout << min(cal(str, s1), cal(str, s2)) << endl;
}

int main(){
    solve();
    return 0;
}

// 移动1或0的次数
int swapTimes(string& s, char c) {
    int times = 0;
    int pos = 0;
    for(int i = 0; i < s.size(); ++i) {
        if(s[i] == c) {
            times += abs(i - pos);
            pos += 2;
        }
    }
    return times;
}

void solve() {
    int cnt0 = 0;
    int cnt1 = 0;
    int ans = 0;
    // 记录1和0的个数
    for(int i = 0; i < s.size(); ++i) {
        if(s[i] == '0') {
            cnt0++;
        }
        else {
            cnt1++;
        }
    }
    // 得到移动0到正确位置需要的次数
    int op0 = swapTimes(s, '0');
    // 得到移动1到正确位置需要的次数
    int op1 = swapTimes(s, '1');
    // 0和1的个数相等则取最小值
    if(cnt0 == cnt1) {
        ans = min(op0, op1);
    }
    else { // 不相等就取其中更大的值
        ans = cnt0 > cnt1 ? op0 : op1;    
    }
    cout << ans << endl;
}


int main()
{
    while(cin >> s) {
        solve();
    }
    return 0;
}

/**
 * 思路：贪心算法
 * 
 * 首先对 01 串进行分析，当字符串长度为奇数时，最终的字符串
 * 一定以比较多的那个字符开头结尾。当字符串长度为偶数时，最终的
 * 字符串开头可能是'1' 也有可能是 '0'。此时，我们已经知道了
 * 目标字符串的样子。
 * 
 * 下一步的问题是如何以最小的次数将当前的字符串通过题目中的
 * 相邻字符相互交换的操作变为目标字符串。
 * 
 * 将一个字符从 i 位置移动到 j 位置，所需最低的次数是abs(i - j)
 * i 是当前位置，j 是最终位置。
 * 
 * 明确了最终的目标字符串，以及移动的次数了以后，选择的贪心策略是
 * 从头开始遍历字符串，选择目标字符，将其移动到最近的目标位置，
 * 记录其花费的次数。
 * 
 * 这个贪心策略的合理性在于，让每个字符都移动到距离它最近的位置，
 * 通过局部最优的移动策略得到总体最优的移动策略。
 * 
 * 以'1'和'0'分别为目标字符遍历一次后，
 * 根据二者的数量关系，确定最终的答案。
 */

import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        String input = scanner.next();
        // 统计字符串中 '0' 和 '1' 的出现次数
        int countZeros = 0;
        int countOnes = 0;
        for (int i = 0; i < input.length(); i++) {
            if (input.charAt(i) == '0') {
                countZeros++;
            } else {
                countOnes++;
            }
        }
        long swapToZero = countSwaps(input, '0');  // 交换 0 所需要的操作次数
        long swapToOne = countSwaps(input, '1');  // 交换 1 所需要的操作次数
        if (countZeros > countOnes) {
            System.out.println(swapToZero);
        } else if (countZeros < countOnes) {
            System.out.println(swapToOne);
        } else {
            System.out.println(Math.min(swapToOne, swapToZero));
        }
    }
    // 计算所需要的交换次数
    public static long countSwaps(String input, char c) {
        int currentSwapIndex = 0;
        long swaps = 0;
        for (int i = 0; i < input.length(); i++) {
            if (input.charAt(i) == c) {
                swaps += Math.abs(i - currentSwapIndex);
                currentSwapIndex += 2;
            }
        }
        return swaps;
    }
}

def swap_n(char_list, char_s):

    swap = 0
    for i in range(len(char_list)):

            current += 2
    return swap
