## Java

```java
import java.util.*;

public class Main {
    public static void main(String[] args) throws IOException {
        Scanner scanner = new Scanner(System.in);
        int n = scanner.nextInt();
        int res = scanner.nextInt();
        for (int i = 1; i < n; i++) {
            res ^= scanner.nextInt();
        }
        System.out.println(res);
    }
}
```

from collections import Counter

length = int(input())

count = Counter(nums)
for key,value in count.items():
    if value == 1:
        print(key)
        break

package main

import(
    "fmt"
)

func main(){
    var (
        ans int 
        n int
    )
    fmt.Scan(&n);
    for n>0{
        var x int 
        fmt.Scan(&x)
        ans^=x;
        n--;
    }
    fmt.Println(ans);
}
