## Js 

```Js 
// 引入readline模块来读取标准输入
const readline = require('readline');

// 创建readline接口
const rl = readline.createInterface({
  input: process.stdin,
  output: process.stdout
});

// 处理输入和输出
function processInput() {
  rl.on('line', (input) => {
    // 将输入按空格分割成a和b的数组
    const [a, b] = input.split(' ').map(Number);
    
    // 计算a和b的和并输出
    const sum = a + b;
    console.log(sum);
  });
}

// 开始处理输入
processInput();
```

## PHP 

```php 
<?php
while (true) {
    $input = fgets(STDIN);
    if ($input === false) {
        break;
    }
    $numbers = explode(' ', trim($input));
    $a = intval($numbers[0]);
    $b = intval($numbers[1]);
    echo ($a + $b) . "\n";
}
```
